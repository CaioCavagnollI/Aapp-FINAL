### Middleware (gating por sessão/plano/role com suporte a `client`)

`apps/web/src/middleware.ts`

```ts
import { NextResponse, type NextRequest } from "next/server";
import { createServerClient } from "@supabase/ssr";

type RouteRule = {
  prefix: string;
  auth?: boolean;
  allowedPlans?: string[];
  allowedRoles?: string[];
};

const ROUTE_RULES: RouteRule[] = [
  { prefix: "/dashboard", auth: true },
  { prefix: "/chat", auth: true, allowedPlans: ["free", "pro", "pro_plus", "b2b", "university"] },
  { prefix: "/prescrever", auth: true, allowedPlans: ["pro", "pro_plus", "b2b", "university"] },
  { prefix: "/scanner", auth: true, allowedPlans: ["pro", "pro_plus", "b2b", "university"] },
  { prefix: "/uploads", auth: true },
  { prefix: "/admin", auth: true, allowedRoles: ["admin"] }
];

function matchRule(pathname: string) {
  return ROUTE_RULES.find((r) => pathname.startsWith(r.prefix));
}

export async function middleware(req: NextRequest) {
  const res = NextResponse.next();
  const pathname = req.nextUrl.pathname;
  const rule = matchRule(pathname);
  if (!rule) return res;

  const supabase = createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    {
      cookies: {
        get(name: string) {
          return req.cookies.get(name)?.value;
        },
        set(name: string, value: string, options: any) {
          res.cookies.set({ name, value, ...options });
        },
        remove(name: string, value: string, options: any) {
          res.cookies.set({ name, value: "", ...options });
        }
      }
    }
  );

  const { data: auth } = await supabase.auth.getUser();
  const user = auth.user;

  if (rule.auth && !user) {
    const url = req.nextUrl.clone();
    url.pathname = "/login";
    url.searchParams.set("next", pathname);
    return NextResponse.redirect(url);
  }

  if (!user) return res;

  const { data: profile } = await supabase
    .from("profiles")
    .select("role, plan_tier")
    .eq("user_id", user.id)
    .single();

  const role = profile?.role ?? "student";
  const planTier = profile?.plan_tier ?? "free";

  if (rule.allowedRoles && !rule.allowedRoles.includes(role)) {
    return NextResponse.redirect(new URL("/dashboard?forbidden=role", req.url));
  }

  // Patch V5.4: role=client ignora bloqueio de plano em rotas liberadas
  if (rule.allowedPlans && !rule.allowedPlans.includes(planTier) && role !== "client") {
    return NextResponse.redirect(new URL("/pricing?forbidden=plan", req.url));
  }

  return res;
}

export const config = {
  matcher: ["/dashboard/:path*", "/chat/:path*", "/prescrever/:path*", "/scanner/:path*", "/uploads/:path*", "/admin/:path*"]
};
```

```tsx
"use client";

import { useState } from "react";
import { createClient } from "@/lib/supabase/client";

export default function LoginPage() {
  const supabase = createClient();
  const [email, setEmail] = useState("");
  const [message, setMessage] = useState("");
  const [loading, setLoading] = useState<string | null>(null);

  async function signInWithGoogle() {
    setLoading("google");
    setMessage("");
    const origin = window.location.origin;
    const { error } = await supabase.auth.signInWithOAuth({
      provider: "google",
      options: { redirectTo: `${origin}/auth/callback` }
    });
    if (error) setMessage(error.message);
    setLoading(null);
  }

  async function signInWithMagicLink(e: React.FormEvent) {
    e.preventDefault();
    setLoading("email");
    setMessage("");
    const origin = window.location.origin;
    const { error } = await supabase.auth.signInWithOtp({
      email,
      options: { emailRedirectTo: `${origin}/auth/callback` }
    });
    if (error) setMessage(error.message);
    else setMessage("Link mágico enviado.");
    setLoading(null);
  }

  return (
    <main className="mx-auto max-w-lg p-6">
      <h1 className="text-2xl font-bold">Entrar</h1>
      <div className="mt-4 space-y-4 rounded-xl border p-4">
        <button onClick={signInWithGoogle} className="w-full rounded border px-3 py-2">
          {loading === "google" ? "Conectando..." : "Entrar com Google"}
        </button>
```

```ts
import { NextResponse } from "next/server";
import { createServerClient } from "@supabase/ssr";
import { cookies } from "next/headers";

export async function GET(req: Request) {
  const url = new URL(req.url);
  const code = url.searchParams.get("code");
  const next = url.searchParams.get("next") ?? "/dashboard";

  const cookieStore = cookies();

  const supabase = createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    {
      cookies: {
        get(name: string) {
          return cookieStore.get(name)?.value;
        },
        set(name: string, value: string, options: any) {
          cookieStore.set({ name, value, ...options });
        },
        remove(name: string, options: any) {
          cookieStore.set({ name, value: "", ...options });
        }
      }
    }
  );

  if (code) {
    await supabase.auth.exchangeCodeForSession(code);

    const { data: auth } = await supabase.auth.getUser();
    if (auth.user) {
      await supabase.from("profiles").upsert(
        {
          user_id: auth.user.id,
          name: auth.user.user_metadata?.full_name ?? auth.user.email ?? "Usuário",
          role: "student",
          plan_tier: "free",
          locale: "pt-BR"
        },
        { onConflict: "user_id" }
      );
    }
  }

  return NextResponse.redirect(new URL(next, url.origin));
}
```

```ts
  return NextResponse.redirect(new URL(next, url.origin));
```

```ts
import { NextResponse } from "next/server";
import { stripe } from "@/lib/stripe";
import { cookies } from "next/headers";
import { createServerSupabaseClient } from "@/lib/supabase/server";

export async function POST(req: Request) {
  const body = await req.json();
  const { priceId, mode = "subscription", successPath = "/dashboard", cancelPath = "/pricing" } = body;

  const supabase = createServerSupabaseClient();
  const { data: auth } = await supabase.auth.getUser();

  if (!auth.user) {
    return NextResponse.json({ error: "unauthorized" }, { status: 401 });
  }

  const cookieStore = cookies();
  const affiliateClickRef = cookieStore.get("fv_aff_ref")?.value ?? null;

  const session = await stripe.checkout.sessions.create({
    mode,
    customer_email: auth.user.email,
    line_items: [{ price: priceId, quantity: 1 }],
    success_url: `${process.env.NEXT_PUBLIC_APP_URL}${successPath}`,
    cancel_url: `${process.env.NEXT_PUBLIC_APP_URL}${cancelPath}`,
    metadata: {
      user_id: auth.user.id,
      ...(affiliateClickRef ? { affiliate_click_ref: affiliateClickRef } : {})
    }
  });

  return NextResponse.json({ url: session.url });
}
```

```ts
import { headers } from "next/headers";
import { NextResponse } from "next/server";
import { stripe } from "@/lib/stripe";

export async function POST(req: Request) {
  const body = await req.text();
  const sig = headers().get("stripe-signature");
  if (!sig) return new NextResponse("Missing stripe-signature", { status: 400 });

  let event: any;
  try {
    event = stripe.webhooks.constructEvent(body, sig, process.env.STRIPE_WEBHOOK_SECRET!);
  } catch (err: any) {
    return new NextResponse(`Webhook Error: ${err.message}`, { status: 400 });
  }

  // Minimal: forward conversion; produção: persist orders/subscriptions/entitlements na API
  switch (event.type) {
    case "checkout.session.completed": {
      const session = event.data.object as any;
      const affiliateClickRef = session.metadata?.affiliate_click_ref ?? null;

      if (affiliateClickRef) {
        try {
          await fetch(`${process.env.NEXT_PUBLIC_API_URL || "http://localhost:4000"}/affiliates/conversion`, {
            method: "POST",
            headers: { "content-type": "application/json" },
            body: JSON.stringify({
              clickRef: affiliateClickRef,
              subscriptionId: session.subscription ?? null,
              grossAmountCents: session.amount_total ?? 0
            })
          });
        } catch {}
      }
      break;
    }
    case "customer.subscription.updated":
    case "customer.subscription.deleted":
    case "invoice.payment_failed":
      break;
  }

  return NextResponse.json({ received: true });
}
```

```ts
import { CanActivate, ExecutionContext, Injectable, UnauthorizedException } from "@nestjs/common";
import { getSupabaseAdmin } from "../../lib/supabase/supabase-admin";

@Injectable()
export class AuthGuard implements CanActivate {
  async canActivate(ctx: ExecutionContext): Promise<boolean> {
    const req = ctx.switchToHttp().getRequest();
    const authHeader = req.headers.authorization as string | undefined;

    if (!authHeader?.startsWith("Bearer ")) throw new UnauthorizedException("Missing bearer token");
    const token = authHeader.slice("Bearer ".length);

    const supabase = getSupabaseAdmin();
    const { data, error } = await supabase.auth.getUser(token);
    if (error || !data.user) throw new UnauthorizedException("Invalid token");

    const { data: profile } = await supabase
      .from("profiles")
      .select("role, plan_tier, institution_id, is_founder")
      .eq("user_id", data.user.id)
      .single();

    req.user = {
      userId: data.user.id,
      role: profile?.role ?? "student",
      planTier: profile?.plan_tier ?? "free",
      institutionId: profile?.institution_id ?? null,
      isFounder: profile?.is_founder ?? false
    };

    return true;
  }
}
```

```ts
import { ForbiddenException, Injectable } from "@nestjs/common";
import type { AuthUserContext } from "@fitversum/contracts";
import { getSupabaseAdmin } from "../supabase/supabase-admin";
import { PLAN_LIMITS } from "../plan/plan-limits";

type UsageKey = "ai_messages_count" | "workout_generations_count" | "scanner_scans_count";

@Injectable()
export class UsageService {
  private getPeriodYM(date = new Date()) {
    const y = date.getUTCFullYear();
    const m = String(date.getUTCMonth() + 1).padStart(2, "0");
    return `${y}-${m}`;
  }

  private getLimitForKey(user: AuthUserContext, key: UsageKey): number | null {
    // Patch V5.4: role=client (sem plano) com quotas próprias
    if (user.role === "client") {
      if (key === "ai_messages_count") return 20;
      if (key === "scanner_scans_count") return null;
      if (key === "workout_generations_count") return 0;
    }

    const plan = PLAN_LIMITS[user.planTier];
    if (!plan) return null;

    if (key === "ai_messages_count") return plan.aiMonthlyMessages;
    if (key === "workout_generations_count") return plan.workoutPrescriptionsMonthly;
    if (key === "scanner_scans_count") return plan.scannerEnabled ? null : 0;
    return null;
  }

  async assertAndIncrement(user: AuthUserContext, key: UsageKey, increment = 1) {
    const supabase = getSupabaseAdmin();
    const periodYM = this.getPeriodYM();

    const { data: row, error } = await supabase
      .from("usage_counters_monthly")
      .select("*")
      .eq("user_id", user.userId)
      .eq("period_ym", periodYM)
      .maybeSingle();

    if (error) throw new Error(error.message);

    const currentRow =
      row ??
      (await supabase
        .from("usage_counters_monthly")
        .insert({ user_id: user.userId, period_ym: periodYM })
        .select("*")
        .single()).data;

    const current = Number(currentRow?.[key] ?? 0);
    const limit = this.getLimitForKey(user, key);

    if (limit !== null && current + increment > limit) {
      throw new ForbiddenException({
        code: "quota_exceeded",
        metric: key,
        current,
        increment,
        limit,
        periodYM
      });
    }

    const patch: Record<string, number> = { [key]: current + increment };

    const { error: updateError } = await supabase
      .from("usage_counters_monthly")
      .update(patch)
      .eq("id", currentRow.id);

    if (updateError) throw new Error(updateError.message);

    return { ok: true, metric: key, current: current + increment, limit, periodYM };
  }

  async getCurrentUsage(user: AuthUserContext) {
    const supabase = getSupabaseAdmin();
    const periodYM = this.getPeriodYM();

    const { data } = await supabase
      .from("usage_counters_monthly")
      .select("*")
      .eq("user_id", user.userId)
      .eq("period_ym", periodYM)
      .maybeSingle();

    return {
      periodYM,
      usage: data ?? {
        ai_messages_count: 0,
        workout_generations_count: 0,
        scanner_scans_count: 0
      }
    };
  }
}
```

### Rate limit (Upstash) — serviço base

`apps/api/src/lib/rate-limit/upstash-rate-limit.service.ts`

```ts
import { Injectable } from "@nestjs/common";
import { Redis } from "@upstash/redis";

@Injectable()
export class UpstashRateLimitService {
  private redis: Redis | null = null;

  private getClient() {
    if (this.redis) return this.redis;
    if (process.env.UPSTASH_REDIS_REST_URL && process.env.UPSTASH_REDIS_REST_TOKEN) {
      this.redis = new Redis({
        url: process.env.UPSTASH_REDIS_REST_URL,
        token: process.env.UPSTASH_REDIS_REST_TOKEN
      });
    }
    return this.redis;
  }

  async hit(key: string, windowSeconds: number, maxHits: number) {
    const redis = this.getClient();
    if (!redis) {
      return { allowed: true, count: 1, remaining: maxHits - 1, resetInSec: windowSeconds, bypass: true };
    }

    const now = Math.floor(Date.now() / 1000);
    const bucketKey = `rl:${key}:${Math.floor(now / windowSeconds)}`;
    const count = await redis.incr(bucketKey);
    if (count === 1) await redis.expire(bucketKey, windowSeconds);

    const allowed = count <= maxHits;
    return { allowed, count, remaining: Math.max(0, maxHits - count), resetInSec: windowSeconds };
  }
}
```

```ts
import crypto from "crypto";
import type { RetrievedChunk } from "./rag-adapter.interface";
import type { AISource } from "@fitversum/contracts";

export function normalizeChunksToSources(chunks: RetrievedChunk[]): AISource[] {
  return chunks.map((c) => ({
    documentId: c.documentId,
    title: c.title,
    author: c.author,
    year: c.year,
    pageStart: c.pageStart,
    pageEnd: c.pageEnd,
    chunkId: c.chunkId,
    score: c.score,
    locator: c.locator
  }));
}

export function makeCitationId(chunk: Pick<RetrievedChunk, "documentId" | "chunkId">) {
  return `cit_${crypto.createHash("sha1").update(`${chunk.documentId}:${chunk.chunkId}`).digest("hex").slice(0, 10)}`;
}

export function buildShortQuote(text: string, maxLen = 180) {
  const clean = text.replace(/\s+/g, " ").trim();
  return clean.length <= maxLen ? clean : `${clean.slice(0, maxLen - 1)}…`;
}
```

```ts
import { Injectable } from "@nestjs/common";
import { getSupabaseAdmin } from "../supabase/supabase-admin";

@Injectable()
export class ExerciseCatalogService {
  async suggestByEquipId(equipId: string, limit = 8) {
    const supabase = getSupabaseAdmin();

    const { data: equipment } = await supabase
      .from("equipment")
      .select("id, equip_id, equipment_name")
      .eq("equip_id", equipId)
      .single();

    if (!equipment) return [];

    const { data, error } = await supabase
      .from("equipment_exercise")
      .select("sort_order, exercise:exercise_id (id, slug, name, exercise_family)")
      .eq("equipment_id", equipment.id)
      .eq("is_contraindicated", false)
      .order("sort_order", { ascending: true })
      .limit(limit);

    if (error) throw new Error(error.message);
    return (data ?? []).map((r: any) => r.exercise).filter(Boolean);
  }

  async musclesByExercises(exerciseIds: string[]) {
    if (!exerciseIds.length) return { primeMovers: [], synergists: [], stabilizers: [] };

    const supabase = getSupabaseAdmin();
    const { data, error } = await supabase
      .from("exercise_muscle")
      .select("exercise_id, muscle_name, role_type")
      .in("exercise_id", exerciseIds);

    if (error) throw new Error(error.message);

    const prime = new Set<string>();
    const syn = new Set<string>();
    const stab = new Set<string>();

    for (const row of data ?? []) {
      if (row.role_type === "prime_mover") prime.add(row.muscle_name);
      if (row.role_type === "synergist") syn.add(row.muscle_name);
      if (row.role_type === "stabilizer") stab.add(row.muscle_name);
    }

    return { primeMovers: [...prime], synergists: [...syn], stabilizers: [...stab] };
  }
}
```

```ts
function budgetsForMode(mode: string, originalChars: number) {
  const pct = (p: number) => Math.floor(originalChars * p);
  if (mode === "simple") return { ai: pct(0.10), advisor: pct(0.20), total: pct(0.30) };
  if (mode === "mentorship" || mode === "translation") return { ai: 0, advisor: 0, total: pct(0.50) };
  if (mode === "translation_total") return { ai: 0, advisor: 0, total: pct(1.10) };
  return { ai: pct(0.10), advisor: pct(0.20), total: pct(0.30) };
}

@Injectable()
export class EditorialService {
  private dmp = new diff_match_patch();

  async createSubmission(userId: string, body: any) {
    const supabase = getSupabaseAdmin();
    const serviceMode = body.serviceMode ?? "simple";

    const { data: sub, error } = await supabase
      .from("editorial_submissions")
      .insert({
        user_id: userId,
        status: "submitted",
        title: body.title ?? null,
        locale: body.locale ?? "pt-BR",
        notes: body.notes ?? null,
        source_asset_id: body.sourceAssetId ?? null,
        service_mode: serviceMode,
        translation_total: Boolean(body.translationTotal ?? false)
      })
      .select("*")
      .single();
    if (error) throw new Error(error.message);

    // Doc stub: em P1.2, extrair texto real do asset
    const originalText = `SUBMISSION_ASSET_REF:${body.sourceAssetId}`;
    const originalCharCount = originalText.length;
    const budgets = budgetsForMode(serviceMode, originalCharCount);

    await supabase.from("submission_documents").insert({
      submission_id: sub.id,
      original_text: originalText,
      current_text: originalText
    });

    await supabase.from("editorial_submissions").update({
      original_char_count: originalCharCount,
      ai_replace_budget: budgets.ai,
      advisor_replace_budget: budgets.advisor,
      total_replace_budget: budgets.total
    }).eq("id", sub.id);

    return { ok: true, submission: sub };
  }

  async listSubmissions(userId: string) {
    const supabase = getSupabaseAdmin();
    const { data, error } = await supabase
      .from("editorial_submissions")
      .select("*")
      .eq("user_id", userId)
      .order("created_at", { ascending: false });
    if (error) throw new Error(error.message);
    return { items: data ?? [] };
  }

  async getSubmission(userId: string, id: string) {
    const supabase = getSupabaseAdmin();
    const { data, error } = await supabase
      .from("editorial_submissions")
      .select("*")
      .eq("id", id)
      .eq("user_id", userId)
      .single();
    if (error) throw new Error(error.message);
    return { item: data };
  }

  async getDocument(userId: string, submissionId: string) {
    const supabase = getSupabaseAdmin();
    await this.getSubmission(userId, submissionId);

    const { data: doc, error } = await supabase
      .from("submission_documents")
      .select("*")
      .eq("submission_id", submissionId)
      .single();
    if (error) throw new Error(error.message);

    const { data: edits } = await supabase
      .from("submission_edits")
      .select("actor_type, removed_chars, created_at")
      .eq("submission_id", submissionId);

    const usedTotal = (edits ?? []).reduce((a: number, e: any) => a + Number(e.removed_chars || 0), 0);

    const { data: sub } = await supabase
      .from("editorial_submissions")
      .select("original_char_count, ai_replace_budget, advisor_replace_budget, total_replace_budget")
      .eq("id", submissionId)
      .single();

    return {
      submissionId,
      originalCharCount: Number(sub?.original_char_count ?? doc.original_text.length),
      budgets: {
        ai: Number(sub?.ai_replace_budget ?? 0),
        advisor: Number(sub?.advisor_replace_budget ?? 0),
        total: Number(sub?.total_replace_budget ?? 0),
        usedAi: 0,
        usedAdvisor: 0,
        usedTotal
      },
      originalText: doc.original_text,
      currentText: doc.current_text
    };
  }

  async applyPatch(userId: string, submissionId: string, body: any) {
    const supabase = getSupabaseAdmin();
    const doc = await this.getDocument(userId, submissionId);

    const patchText = String(body.patchText ?? "");
    if (!patchText.trim()) throw new Error("empty_patch");

    // PATCH STRATEGY (MVP): replace entire doc with patchText (budget counts deletions)
    const diffs = this.dmp.diff_main(doc.currentText, patchText);
    this.dmp.diff_cleanupSemantic(diffs);

    const removedChars = diffs
      .filter((d) => d[0] === -1)
      .reduce((a, d) => a + (d[1]?.length ?? 0), 0);

    const newUsedTotal = doc.budgets.usedTotal + removedChars;
    if (doc.budgets.total > 0 && newUsedTotal > doc.budgets.total) {
      throw new ForbiddenException({ code: "total_budget_exceeded", used: newUsedTotal, limit: doc.budgets.total });
    }

    await supabase.from("submission_documents").update({ current_text: patchText }).eq("submission_id", submissionId);

    await supabase.from("submission_edits").insert({
      submission_id: submissionId,
      actor_type: body.actorType === "advisor" ? "advisor" : "ai",
      patch: { mode: "replace_all" },
      removed_chars: removedChars
    });

    return {
      ok: true,
      removedChars,
      budgets: { ...doc.budgets, usedTotal: newUsedTotal },
      currentText: patchText
    };
  }

  async downloadDeliverable(userId: string, deliverableId: string) {
    // P0 roadmap: implementar signed URL real aqui
    // MVP: apenas retorna placeholder
    await this.assertDeliverableOwnership(userId, deliverableId);
    return { signedUrl: "SIGNED_URL_PLACEHOLDER", expiresInSec: 600 };
  }

  private async assertDeliverableOwnership(userId: string, deliverableId: string) {
    const supabase = getSupabaseAdmin();
    const { data, error } = await supabase
      .from("editorial_deliverables")
      .select("id, submission_id")
      .eq("id", deliverableId)
      .single();
    if (error) throw new Error(error.message);
    const subId = data.submission_id;

    const { data: sub } = await supabase
      .from("editorial_submissions")
      .select("user_id")
      .eq("id", subId)
      .single();
    if (!sub || sub.user_id !== userId) throw new ForbiddenException("not_owner");
  }
}
```

## 15) Worker (BullMQ) — `apps/api/src/worker.ts` (scaffold)

```ts
import "reflect-metadata";
import { Worker } from "bullmq";
import IORedis from "ioredis";
import { getSupabaseAdmin } from "./lib/supabase/supabase-admin";

const connection = process.env.REDIS_URL ? new IORedis(process.env.REDIS_URL) : undefined;

type JobData = { jobType: string; payload: any };

function requireRedis() {
  if (!connection) throw new Error("REDIS_URL not set");
  return connection;
}

async function updateJob(table: "ingestion_jobs" | "editorial_review_jobs", id: string, patch: any) {
  const supabase = getSupabaseAdmin();
  await supabase.from(table).update(patch).eq("id", id);
}

async function run() {
  const conn = requireRedis();

  const worker = new Worker<JobData>(
    "fitversum_jobs",
    async (job) => {
      const { jobType, payload } = job.data;

      // payload deve carregar { jobId, table } etc.
      if (payload?.table && payload?.jobId) {
        await updateJob(payload.table, payload.jobId, { status: "running", progress: 5, updated_at: new Date().toISOString() });
      }

      // Ingestion (P1): rag_parse/rag_chunk/rag_embed/rag_index
      // Editorial (P1): ai_precheck/human_review/translation_precheck
      // MVP: finalizar
      if (payload?.table && payload?.jobId) {
        await updateJob(payload.table, payload.jobId, { status: "done", progress: 100, updated_at: new Date().toISOString(), result: { ok: true } });
      }

      return { ok: true };
    },
    { connection: conn as any }
  );

  worker.on("failed", async (job, err) => {
    if (!job) return;
    const payload: any = job.data?.payload;
    if (payload?.table && payload?.jobId) {
      await updateJob(payload.table, payload.jobId, { status: "failed", error: String(err?.message ?? err), updated_at: new Date().toISOString() });
    }
  });

  console.log("✅ Worker listening on queue fitversum_jobs");
}

run().catch((e) => {
  console.error("❌ Worker failed:", e);
  process.exit(1);
});
```

```ts
const API = process.env.EXPO_PUBLIC_API_URL || "http://localhost:4000";

export async function apiPost(path: string, body: any) {
  const { data } = await supabase.auth.getSession();
  const token = data.session?.access_token;
  if (!token) throw new Error("no_session");

  const r = await fetch(`${API}${path}`, {
    method: "POST",
    headers: { "content-type": "application/json", authorization: `Bearer ${token}` },
    body: JSON.stringify(body)
  });
  if (!r.ok) throw new Error(await r.text());
  return r.json();
```
